def consulta_metadatos_campos(db_name,nombre_tabla):
    query_metadatos = f"""SELECT * FROM {db_name}.[dbo].[Metadata_Campos] t
    where t.Tabla = '{nombre_tabla}'
    ORDER BY t.Tabla
    """
    # Ejecuta la consulta
    cursor.execute(query_metadatos)
    # Obtén los resultados de la consulta como lista de listas
    rows = [list(row) for row in cursor.fetchall()]  # Convertir cada tupla en lista
    columns = [column[0] for column in cursor.description]  # Obtener nombres de las columnas
    # Crear el DataFrame con los resultados
    df = pd.DataFrame(rows, columns=columns)
    return df

def lista_campos_repetidos(db_name_final):
    query_fields_list = f"""SELECT 
            TABLE_CATALOG AS Base,
            TABLE_SCHEMA AS Esquema,
            TABLE_NAME AS Tabla,
            COLUMN_NAME AS Columna
        FROM  {db_name_final}.INFORMATION_SCHEMA.COLUMNS
        WHERE 
            TABLE_CATALOG = 'COL_CTY_PARAMETRICA'
            AND TABLE_SCHEMA = 'dbo'
            AND TABLE_NAME NOT IN ('Metadata_Tablas','Metadata_Campos','Metadata_Relaciones','Fact_Metadatos_Tablas','Fact_Metadatos_Campos');"""
    # Ejecutar la consulta
    cursor.execute(query_fields_list)

    # Obtén los resultados de la consulta como lista de listas
    rows = [list(row) for row in cursor.fetchall()]  # Convertir cada tupla en lista
    columns = [column[0] for column in cursor.description]  # Obtener nombres de las columnas
    # Crear el DataFrame con los resultados
    df_campos = pd.DataFrame(rows, columns=columns)

    # Contar las ocurrencias de cada valor en la columna "Columna"
    conteo = df_campos['Columna'].value_counts()
    # Filtrar las columnas que aparecen más de una vez
    columnas_repetidas = conteo[conteo > 1].index
    # Crear un dataframe con las columnas repetidas y sus conteos
    df_campos_repetidos = pd.DataFrame({
        'Campo': columnas_repetidas,
        #'Frecuencia': conteo[conteo > 1].values
        'Campo_Repetido': 'SI'
    })
    df_campos_repetidos
    return df_campos_repetidos

def fact_metadatos_campos(db_name_final,nombre_tabla):
    #Consultar Metadatos de los campos e identificar si estan completos
    df_metadatos_campos = consulta_metadatos_campos(db_name_final,nombre_tabla)
    df_metadatos_campos['Metadatos_Completos'] = df_metadatos_campos.isnull().any(axis=1).apply(lambda x: 'NO' if x else 'SI')

    # Consulta la tabla
    query = f"SELECT * FROM {db_name}.{esquema}.{nombre_tabla};"

    # Ejecuta la consulta
    cursor.execute(query)

    # Obtén los resultados de la consulta como lista de listas
    rows = [list(row) for row in cursor.fetchall()]  # Convertir cada tupla en lista
    columns = [column[0] for column in cursor.description]  # Obtener nombres de las columnas

    # Crear el DataFrame con los resultados
    df = pd.DataFrame(rows, columns=columns)

    # Reemplazar espacios en blanco por NaN
    df.replace(r'^\s*$', np.nan, regex=True, inplace=True)

    # Contar los registros no nulos por columna
    conteo_por_columna = df.count()

    # Convertir el conteo a un dataframe
    df_conteo = conteo_por_columna.reset_index()
    df_conteo.columns = ['Campo', 'No_Nulos']  # Renombrar las columnas para mayor claridad
    df_conteo['TotalRegistros'] = len(df)
    df_conteo['Base'] = db_name
    df_conteo['Esquema'] = esquema
    df_conteo['Tabla'] = nombre_tabla
    df_conteo['Nulos'] = df_conteo['TotalRegistros'] - df_conteo['No_Nulos']

    #Incluir informacion de los campos repetidos
    df_campos_repetidos = lista_campos_repetidos(db_name_final)
    df_totales = pd.merge(df_conteo,df_campos_repetidos, how ='left', on = 'Campo')
    df_totales['Campo_Repetido'] = df_totales['Campo_Repetido'].fillna('NO')

    df_totales = df_totales[['Base', 'Esquema', 'Tabla','Campo', 'TotalRegistros', 'Nulos','Campo_Repetido']].copy()

    ##Unir ambos dataframes
    df_metadatos = pd.merge(df_totales,df_metadatos_campos, how ='left', on = ['Base', 'Esquema', 'Tabla','Campo'] )
    ## Regla: Si los campos no tienen metadatos para el tipo de campo se consideran como varchar
    df_metadatos['DataType'] = df_metadatos['DataType'].fillna('varchar')


    lista_campos = list(df_metadatos[['Campo', 'DataType']].itertuples(index=False, name=None))
    df_tipo = []
    for campo,datatype in lista_campos:
        #print(campo,datatype,tipos_datos[datatype])
        tipo = tipos_datos[datatype]
        # Verificar si todos los valores en la columna son del tipo de dato definido
        val_tipo = df[campo].apply(lambda x: True if pd.isnull(x) else (True if tipo == 'image' else isinstance(x, eval(tipo))))
        #print(val_tipo)
        # Contar los valores que son False en val_tipo
        conteo_false = (~val_tipo).sum()
        df_tipo.append(conteo_false)
        # Mostrar el conteo
        #print(f"Cantidad de valores que son False: {conteo_false}")
    df_metadatos['Registros_Sin_Exactitud'] = df_tipo
    df_metadatos['Longitud'] = pd.to_numeric(df_metadatos['Longitud'], errors='coerce').astype('Int64')
    df_metadatos['Metadatos_Completos'] = df_metadatos['Metadatos_Completos'].fillna('NO')
    df_metadatos

    return df_metadatos

# Conectar a las bases y poblar tablas
esquema = 'dbo'
for db_name, conn_str in connections_start.items():
    print(f"Procesando base de datos: {db_name}")
    log.write(f"{time.strftime('%Y-%m-%d %H:%M:%S')} - INFO: Procesando base de datos: {db_name}\n")
    #print(conn_str)
    try:
        # Conexion a base fuente
        conn = pyodbc.connect(conn_str)
        cursor = conn.cursor()

        #Conexion a base destino

        db_name_final = db_final(db_name)
        conn_str_final = connections_final[db_name_final]
        conn_final = pyodbc.connect(conn_str_final)
        cursor_final = conn_final.cursor()
        print("Intentando conexion")
        log.write(f"{time.strftime('%Y-%m-%d %H:%M:%S')} - INFO: Conexión a {db_name_final} exitosa \n")
        print("Conexion Hecha")

        try:
            clean_query = f"DELETE FROM dbo.Fact_Metadatos_Campos;"
            cursor_final.execute(clean_query)
            conn_final.commit()
            print(f"Tabla dbo.Fact_Metadatos_Campos limpiada exitosamente")
            log.write(f"{time.strftime('%Y-%m-%d %H:%M:%S')} - INFO: Tabla dbo.Fact_Metadatos_Campos limpiada exitosamente\n")
            #sql_script += f"-- Limpiar tabla {nombre_tabla_sql}\n" + clean_query + "\n"
        except Exception as e:
            print(f"Error al limpiar la tabla dbo.Fact_Metadatos_Campos: {e}")
            log.write(f"{time.strftime('%Y-%m-%d %H:%M:%S')} - ERROR: Al limpiar la tabla dbo.Fact_Metadatos_Campos: {e}\n")

        tablas_filtradas = lista_tablas(db_name_final,cursor_final)
        for nombre_tabla in tablas_filtradas:
            log.write(f"{time.strftime('%Y-%m-%d %H:%M:%S')} - INFO: Procesando {nombre_tabla} \n")
            df = fact_metadatos_campos(db_name_final,nombre_tabla)
            df['Create_date'] = df['Create_date'].astype('object')
            df['Modify_date'] = df['Modify_date'].astype('object')
            for index, row in df.iterrows():
                # Reemplazar el valor en el DataFrame original con el formato CONVERT
                # Para 'create_date'
                if pd.isna(row['Create_date']):
                    df.at[index, 'Create_date'] = None
                else:
                    df.at[index, 'Create_date'] = f"CONVERT(DATETIME2, '{row['Create_date']}', 120)"     
                # Para 'modify_date'
                if pd.isna(row['Modify_date']):
                    df.at[index, 'Modify_date'] = None
                else:
                    df.at[index, 'Modify_date'] = f"CONVERT(DATETIME2, '{row['Modify_date']}', 120)"      

            for index, row in df.iterrows():
                valores = [
                    "NULL" if pd.isna(valor) else f"'{valor}'"
                    for valor in row
                ]
                valores_sin_comillas = [v.strip("'") if v.startswith("'CONVERT") and v.endswith(")'") else v for v in valores]       
                columnas = ', '.join([f'{col}' for col in df.columns])
                #columnas = {', '.join(df.columns)}
                try:
                    query = f"INSERT INTO dbo.Fact_Metadatos_Campos ({columnas}) VALUES ({', '.join(valores_sin_comillas)})"
                    cursor_final.execute(query)
                    conn_final.commit()
                except Exception as e:
                    print(f"Error en la estructura de archivo de metadatos {esquema}.{nombre_tabla}: {e}")
                    log.write(f"{time.strftime('%Y-%m-%d %H:%M:%S')} - ERROR: En la estructura de archivo de metadatos {esquema}.{nombre_tabla}: {e}\n")
                    continue



    except Exception as e:
        print(f"Error procesando la base de datos {db_name}: {e}")
        log.write(f"{time.strftime('%Y-%m-%d %H:%M:%S')} - ERROR: Procesando la base de datos {db_name}: {e}\n")
