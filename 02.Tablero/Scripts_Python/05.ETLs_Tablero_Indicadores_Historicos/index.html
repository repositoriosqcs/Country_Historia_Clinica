def consulta_fact_metadatos_campos(db_name_final):
    query_metadatos = f"""SELECT * FROM {db_name_final}.[dbo].[Fact_Metadatos_Campos] t"""
    # Ejecuta la consulta
    cursor.execute(query_metadatos)
    # Obtén los resultados de la consulta como lista de listas
    rows = [list(row) for row in cursor.fetchall()]  # Convertir cada tupla en lista
    columns = [column[0] for column in cursor.description]  # Obtener nombres de las columnas
    # Crear el DataFrame con los resultados
    df = pd.DataFrame(rows, columns=columns)
    return df

def exactitud_tabla(db_name_final):
    # Consulta la tabla de fact metadatos campos del codigo 03
    df = consulta_fact_metadatos_campos(db_name_final)

    # Agregar id del indicador y calcularlo
    df['Id_Componente'] = 1
    df['Valor_Indicador'] = 1-df['Registros_Sin_Exactitud']/df['TotalRegistros']
    df['Valor_Indicador'] = df['Valor_Indicador'].fillna(1)

    #Seleccionar las columnas para agrupar
    df_exactitud = df[['Id_Componente','Base','Esquema','Tabla','Valor_Indicador']].copy()

    # Agrupar por las columnas 'Id_Componente','Base', 'Esquema', 'Tabla',  y calcular el promedio
    df_promedio = df_exactitud.groupby(['Id_Componente','Base', 'Esquema', 'Tabla'], as_index=False).mean()
    return df_promedio

def consistencia_tabla(db_name_final):
    # Consulta la tabla de fact metadatos campos del codigo 03
    df = consulta_fact_metadatos_campos(db_name_final)

    #Seleccionar las columnas para agrupar
    df_v1_total = df[['Base','Esquema','Tabla','TotalRegistros']].copy()
    # Agrupar por las columnas 'Base', 'Esquema', 'Tabla',  y calcular la suma
    df_v1_total_sum = df_v1_total.groupby(['Base', 'Esquema', 'Tabla'], as_index=False).sum()

    #Seleccionar las columnas repetidas
    df_v1_repetidos_inicial = df[df['Campo_Repetido'] == 'SI'].copy()
    #Seleccionar las columnas para agrupar
    df_v1_repetidos = df_v1_repetidos_inicial[['Base','Esquema','Tabla','TotalRegistros']].copy()
    # Agrupar por las columnas 'Base', 'Esquema', 'Tabla',  y calcular la suma
    df_v1_repetidos_sum = df_v1_repetidos.groupby(['Base', 'Esquema', 'Tabla'], as_index=False).sum()
    df_v1_repetidos_sum.rename(columns = {'TotalRegistros' : 'Repetidos'},inplace = True)

    ## Unificar para calcular variable V1 (Columnas) para el indicador de consistencia
    df_v1 = pd.merge(df_v1_total_sum,df_v1_repetidos_sum, how ='left', on = ['Base','Esquema','Tabla'] )
    df_v1['Repetidos'] = df_v1['Repetidos'].fillna(0)
    df_v1['V1_Consistencia'] = 1- df_v1['Repetidos']/df_v1['TotalRegistros'] 
    df_v1['V1_Consistencia'] = df_v1['V1_Consistencia'].fillna(1)
    df_v1 = df_v1[['Base','Esquema','Tabla','V1_Consistencia']].copy()

    # Consultar metadatos de tablas
    df_tablas = consulta_fact_metadatos_tablas(db_name_final)
    df_v2 = df_tablas[['Base','Esquema','Tabla','TotalRegistros','Duplicados']].copy()
    ## Calcular variable V2 (Filas) para el indicador de consistencia
    df_v2['V2_Consistencia'] = 1- df_v2['Duplicados']/df_v2['TotalRegistros']
    df_v2['V2_Consistencia'] = df_v2['V2_Consistencia'].fillna(1)
    df_v2 = df_v2[['Base','Esquema','Tabla','V2_Consistencia']].copy()

    ## Unificar y calcular consistencia a nivel de tabla
    df_consistencia = pd.merge(df_v1,df_v2,how ='inner',on = ['Base','Esquema','Tabla'])
    # Agregar id del indicador y calcularlo
    df_consistencia['Id_Componente'] = 3
    df_consistencia['Valor_Indicador'] = (df_consistencia['V1_Consistencia'] + df_consistencia['V2_Consistencia'])*0.5
    df_consistencia['Valor_Indicador'] = df_consistencia['Valor_Indicador'].fillna(1)
    #Seleccionar las columnas para agrupar
    df_consistencia = df_consistencia[['Id_Componente','Base','Esquema','Tabla','Valor_Indicador']].copy()
    return df_consistencia

def trazabilidad_tabla(db_name_final):
    # Consultar metadatos de tablas
    df_tablas = consulta_fact_metadatos_tablas(db_name_final)
    df_tablas_fechas = df_tablas[['Base','Esquema','Tabla','create_date','modify_date','last_user_update']].copy()

    #Identificar si se cuenta con fechas de actualizacion de datos, creacion de tabla y actualizacion de metadatos
    df_tablas_fechas['Actualizacion'] = df_tablas_fechas['last_user_update'].apply(lambda x: 0 if pd.isnull(x) else 1)
    df_tablas_fechas['Creacion'] = df_tablas_fechas['create_date'].apply(lambda x: 0 if pd.isnull(x) else 1)
    df_tablas_fechas['Metadatos'] = df_tablas_fechas['modify_date'].apply(lambda x: 0 if pd.isnull(x) else 1)

    # Calcular Trazabilidad 
    df_tablas_fechas['Trazabilidad_aux'] = df_tablas_fechas['Actualizacion']+df_tablas_fechas['Creacion']+df_tablas_fechas['Metadatos']
    df_tablas_fechas['Valor_Indicador'] = df_tablas_fechas['Trazabilidad_aux'].apply(lambda x: 1 if x > 2 else 0)
    df_tablas_fechas['Id_Componente'] = 5

    df_trazabilidad = df_tablas_fechas[['Id_Componente','Base','Esquema','Tabla','Valor_Indicador']].copy()
    return df_trazabilidad

def conformidad_tabla(db_name_final):
    # Consulta la tabla de fact metadatos campos del codigo 03 
    df_campos = consulta_fact_metadatos_campos(db_name_final)
    df_campos = df_campos[['Base', 'Esquema', 'Tabla', 'Campo', 'Descripcion_Campo', 'DataType','Longitud', 'Create_date', 'Modify_date']].copy()

    # Consultar metadatos de tablas
    df_tablas = consulta_fact_metadatos_tablas(db_name_final)
    df_tablas_descripcion = df_tablas[['Base','Esquema','Tabla','Descripcion_Tabla']].copy()

    #Unificar metadatos
    df_conformidad = pd.merge(df_campos,df_tablas_descripcion, how ='left', on= ['Base','Esquema','Tabla'] )

    # Reemplazar espacios en blanco por NaN
    df_conformidad.replace(r'^\s*$', np.nan, regex=True, inplace=True)

    #Identificar si se cuenta con fechas de actualizacion de datos, creacion de tabla y actualizacion de metadatos
    df_conformidad['Descripcion_Campo_Conformidad'] = df_conformidad['Descripcion_Campo'].apply(lambda x: 0 if pd.isnull(x) else 1)
    df_conformidad['DataType_Conformidad'] = df_conformidad['DataType'].apply(lambda x: 0 if pd.isnull(x) else 1)
    df_conformidad['Longitud_Conformidad'] = df_conformidad['Longitud'].apply(lambda x: 0 if pd.isnull(x) else 1)
    df_conformidad['Create_date_Conformidad'] = df_conformidad['Create_date'].apply(lambda x: 0 if pd.isnull(x) else 1)
    df_conformidad['Modify_date_Conformidad'] = df_conformidad['Modify_date'].apply(lambda x: 0 if pd.isnull(x) else 1)
    df_conformidad['Descripcion_Tabla_Conformidad'] = df_conformidad['Descripcion_Tabla'].apply(lambda x: 0 if pd.isnull(x) else 1)
    df_conformidad['Base_Conformidad'] = df_conformidad['Base'].apply(lambda x: 0 if pd.isnull(x) else 1)
    df_conformidad['Esquema_Conformidad'] = df_conformidad['Esquema'].apply(lambda x: 0 if pd.isnull(x) else 1)
    df_conformidad['Tabla_Conformidad'] = df_conformidad['Tabla'].apply(lambda x: 0 if pd.isnull(x) else 1)
    df_conformidad['Campo_Conformidad'] = df_conformidad['Campo'].apply(lambda x: 0 if pd.isnull(x) else 1)

    # Calcular Conformidad 
    df_conformidad['Valor_Indicador'] = (df_conformidad['Descripcion_Campo_Conformidad'] + df_conformidad['DataType_Conformidad'] 
                                     + df_conformidad['Longitud_Conformidad']+ df_conformidad['Create_date_Conformidad']
                                     + df_conformidad['Modify_date_Conformidad'] + df_conformidad['Descripcion_Tabla_Conformidad']
                                     + df_conformidad['Base_Conformidad'] + df_conformidad['Esquema_Conformidad']
                                     + df_conformidad['Tabla_Conformidad'] + df_conformidad['Campo_Conformidad'])/10
    df_conformidad['Id_Componente'] = 6
    df_conformidad = df_conformidad[['Id_Componente','Base','Esquema','Tabla','Valor_Indicador']].copy()

    # Agrupar por las columnas 'Id_Componente','Base', 'Esquema', 'Tabla',  y calcular el promedio
    df_promedio = df_conformidad.groupby(['Id_Componente','Base', 'Esquema', 'Tabla'], as_index=False).mean()

    return df_promedio
