    03 ETLs Tablero Tablas¶

    El objetivo de este script es generar procesar los metadatos de las tablas y calcular los indicadores de calidad asociados con la estructura necesaria para las visualizaciones del tablero

    Librerías¶

    Para empezar se importan las librerías requeridas, las cadenas de conexión connections_start y connections_final, y las funciones db_final y lista_tablas de Funciones Generales

    import pyodbc
import pandas as pd
import os
import time
from datetime import datetime
from funciones_generales import connections_start,connections_final,db_final,lista_tablas

    Funciones para metadatos de tablas¶

    Para el procesamiento de los metadatos se definen las siguientes funciones que permiten hacer las consultas a las base del repositorio, es decir a COL_CTY_PARAMETRICA y CTY_PRI_PARAMETRICA.:

    # Consulta el número total de registros por tabla y sus duplicados
def totales_tabla(esquema,nombre_tabla):
    # Consultar la tabla
    query = f"SELECT * FROM {esquema}.{nombre_tabla};"

    # Ejecutar la consulta
    cursor.execute(query)
    # Obtener los resultados de la consulta como lista de listas
    rows = [list(row) for row in cursor.fetchall()]  # Convertir cada tupla en lista
    columns = [column[0] for column in cursor.description]  # Obtener nombres de las columnas
    # Crear el DataFrame con los resultados
    df = pd.DataFrame(rows, columns=columns)
    # Calcular número total de registros
    total_registros = len(df)

    # Identificar duplicados
    duplicados = df[df.duplicated()]  # Filtra los duplicados
    numero_duplicados = len(duplicados)

    # Crear un nuevo DataFrame con la información
    resumen_df = pd.DataFrame({
        'Base' : [db_name],
        'Esquema' :[esquema],
        'Tabla' : [nombre_tabla],
        'TotalRegistros': [total_registros],
        'Duplicados': [numero_duplicados]
    })
    return resumen_df

    # Consulta los metadatos de las tablas guardados en la tabla Metadata_Tablas
def consulta_metadatos_tablas(db_name,nombre_tabla):
    # Consultar metadatos de las tablas
    query_metadatos = f"""SELECT * FROM {db_name}.[dbo].[Metadata_Tablas] t
    where t.Tabla = '{nombre_tabla}'
    ORDER BY t.Tabla
    """
    # Ejecutar la consulta
    cursor.execute(query_metadatos)
    # Obtener los resultados de la consulta como lista de listas
    rows = [list(row) for row in cursor.fetchall()]  # Convertir cada tupla en lista
    columns = [column[0] for column in cursor.description]  # Obtener nombres de las columnas
    # Crear el DataFrame con los resultados
    df = pd.DataFrame(rows, columns=columns)
    return df

    # Consulta la frecuencia y fecha de actualizacion por tabla
def datos_actualizacion_tablas(db_name_final,nombre_tabla):
    # Consultar metadatos asociados
    query_metadatos = f"""SELECT DISTINCT
       [Base]
      ,[Esquema]
      ,[Tabla]
      ,[FrecuenciaActualizacion]
      ,[FechaActualizacion]
    FROM [{db_name_final}].[dbo].[Metadata_Tablas]
    where Tabla = '{nombre_tabla}'
    """
    # Ejecutar la consulta
    cursor_final.execute(query_metadatos)
    # Obtener los resultados de la consulta como lista de listas
    rows = [list(row) for row in cursor_final.fetchall()]  # Convertir cada tupla en lista
    columns = [column[0] for column in cursor_final.description]  # Obtener nombres de las columnas
    # Crear el DataFrame con los resultados
    df = pd.DataFrame(rows, columns=columns)
    return df

    # Calcula la fecha de la última actualización requerida
def calcular_ultima_actualizacion(fecha_actualizacion):
    try:
        # Obtener la fecha actual
        fecha_actual = datetime.now()
        # Convertir "01-FEBRERO" en una fecha completa
        dia, mes_texto = fecha_actualizacion.split('-')
        meses = {
            'ENERO': 1, 'FEBRERO': 2, 'MARZO': 3, 'ABRIL': 4, 'MAYO': 5, 'JUNIO': 6,
            'JULIO': 7, 'AGOSTO': 8, 'SEPTIEMBRE': 9, 'OCTUBRE': 10, 'NOVIEMBRE': 11, 'DICIEMBRE': 12
        }
        mes = meses[mes_texto.upper()]

        # Fecha del 1 de febrero en el año actual
        fecha_actualizacion = datetime(fecha_actual.year, mes, int(dia))

        # Comparar fechas
        if fecha_actual >= fecha_actualizacion:
            # Usar el año actual
            return fecha_actualizacion
        else:
            # Usar el año anterior
            return fecha_actualizacion.replace(year=fecha_actual.year - 1)
    except:
        # Fecha del 1 de febrero en el año actual
        fecha_actualizacion = datetime(fecha_actual.year, 2, 1)

        # Comparar fechas
        if fecha_actual >= fecha_actualizacion:
            # Usar el año actual
            return fecha_actualizacion
        else:
            # Usar el año anterior
            return fecha_actualizacion.replace(year=fecha_actual.year - 1)

    # Calcula si la última actualización realizada se encuentra vigente de acuerdo con la fecha de la última actualización requerida
def es_actualizacion_vigente(row):
    if pd.isnull(row['last_user_update']) or pd.isnull(row['Ultima_Actualizacion_Requerida']):
        return 'NO'  # Considerar no vigente si hay valores nulos
    elif row['last_user_update'] is None:
        return 'NO'    
    return 'SI' if row['last_user_update'] >= row['Ultima_Actualizacion_Requerida'] else 'NO'

    # Definición de los grupos de las tablas
dict_descripcion = {
    "ACF" : "Módulo de Activos Fijo",
    "ADP" : "Módulo de Nomina en SAHI",
    "ASI" : "Modulo del Administrador del sistema",
    "ADM" : "Todo lo que tiene que ver con parte administrativo",
    "HCE" : "Todo lo que tiene que ver con temas asistenciales",
    "GEN" : "Todo lo relacionado con datos maestros" 
}

    # Calcula los metadatos de las tablas con la estructura para el tablero
def fact_metadatos_tabla(db_name,esquema,nombre_tabla,df_total):
    # Consultar metadatos disponibles
    df_metadatos = consulta_metadatos_tablas(db_name,nombre_tabla)
    # Identificar si los metadatos están completos
    df_metadatos['Metadatos_Completos'] = df_metadatos.isnull().any(axis=1).apply(lambda x: 'NO' if x else 'SI')
    df_procesado = pd.merge(df_total,df_metadatos,how='left', on =['Base','Esquema','Tabla'])
    # Incluir información de grupos
    df_procesado['Grupo'] = nombre_tabla[:3]
    df_procesado['Descripcion_Grupo'] = df_procesado['Grupo'].apply(
        lambda grupo: dict_descripcion.get(grupo, "Grupo sin descripción asignada")
    )

    df_tabla = df_procesado.copy()
    # Ajustar Fecha de actualizacion
    df_tabla['FechaActualizacion'] = df_tabla['FechaActualizacion'].fillna('01-FEBRERO')
    df_tabla['FechaActualizacion'] = df_tabla['FechaActualizacion'].astype(str)
    # Completa valores nulos
    df_tabla['Metadatos_Completos'] = df_tabla['Metadatos_Completos'].fillna('NO')
    # Ajustar frencuencua de actualizacion 
    df_tabla['FrecuenciaActualizacion'] = df_tabla['FrecuenciaActualizacion'].fillna('ANUAL')
    #Calcular la fecha de la ultima actualizacion requerida
    df_tabla['Ultima_Actualizacion_Requerida'] = df_tabla['FechaActualizacion'].apply(calcular_ultima_actualizacion)

    # Calcular si la ultima actualizacion esta vigente o no
    df_tabla['Actualizacion_Vigente'] = df_tabla.apply(es_actualizacion_vigente, axis=1)

    # Ordenar el dataframe por 'last_user_update' en orden descendente
    df_tabla = df_tabla.sort_values(by='last_user_update', ascending=False)
    df_tabla = df_tabla.drop_duplicates(subset=['Tabla']).copy()
    return df_tabla

    Proceso principal¶

    Se define la conexión a las bases de origen con las cadenas de conexión definidas en connections_start y se ejecuta el siguiente bloque:

    # Conectar a las bases y poblar tablas
esquema = 'dbo'
for db_name, conn_str in connections_start.items():
    print(f"Procesando base de datos: {db_name}")
    log.write(f"{time.strftime('%Y-%m-%d %H:%M:%S')} - INFO: Procesando base de datos: {db_name}\n")
    #print(conn_str)
    try:
        # Conexion a base fuente
        conn = pyodbc.connect(conn_str)
        cursor = conn.cursor()

        #Conexion a base destino

        db_name_final = db_final(db_name)
        conn_str_final = connections_final[db_name_final]
        conn_final = pyodbc.connect(conn_str_final)
        cursor_final = conn_final.cursor()
        print("Intentando conexion")
        log.write(f"{time.strftime('%Y-%m-%d %H:%M:%S')} - INFO: Conexión a {db_name_final} exitosa \n")
        print("Conexion Hecha")

        try:
            # Limpiar la tabla Fact_Metadatos_Tablas
            clean_query = f"DELETE FROM dbo.Fact_Metadatos_Tablas;"
            cursor_final.execute(clean_query)
            conn_final.commit()
            print(f"Tabla dbo.Fact_Metadatos_Tablas limpiada exitosamente")
            log.write(f"{time.strftime('%Y-%m-%d %H:%M:%S')} - INFO: Tabla dbo.Fact_Metadatos_Tablas limpiada exitosamente\n")
        except Exception as e:
            print(f"Error al limpiar la tabla dbo.Fact_Metadatos_Tablas: {e}")
            log.write(f"{time.strftime('%Y-%m-%d %H:%M:%S')} - ERROR: Al limpiar la tabla dbo.Fact_Metadatos_Tablas: {e}\n")

        # Identificar tablas a incluir
        tablas_filtradas = lista_tablas(db_name_final,cursor_final)
        for nombre_tabla in tablas_filtradas:
            log.write(f"{time.strftime('%Y-%m-%d %H:%M:%S')} - INFO: Procesando {nombre_tabla} \n")
            # Calcular totales
            df_total = totales_tabla(esquema,nombre_tabla)
            # Completar todos los metadatos
            df = fact_metadatos_tabla(db_name_final,esquema,nombre_tabla,df_total)
            # Ajustar formtatos de campos de fecha
            df['create_date'] = df['create_date'].astype('object')
            df['modify_date'] = df['modify_date'].astype('object')
            df['last_user_update'] = df['last_user_update'].astype('object')
            df['Ultima_Actualizacion_Requerida'] = df['Ultima_Actualizacion_Requerida'].astype('object')
            # Ajustar estructura para subir por medio de comandos SQL
            for index, row in df.iterrows():
                # Reemplazar el valor en el DataFrame original con el formato CONVERT
                # Para 'create_date'
                if pd.isna(row['create_date']):
                    df.at[index, 'create_date'] = None
                else:
                    df.at[index, 'create_date'] = f"CONVERT(DATETIME2, '{row['create_date']}', 120)"     
                # Para 'modify_date'
                if pd.isna(row['modify_date']):
                    df.at[index, 'modify_date'] = None
                else:
                    df.at[index, 'modify_date'] = f"CONVERT(DATETIME2, '{row['modify_date']}', 120)"      

                # Para 'last_user_update' 
                if pd.isna(row['last_user_update']):
                    df.at[index, 'last_user_update'] = None
                else:
                    df.at[index, 'last_user_update'] = f"CONVERT(DATETIME2, '{row['last_user_update']}', 120)"

                # Para 'Ultima_actualizacion_requerida' (manejo con try-except)
                if pd.isna(row['Ultima_Actualizacion_Requerida']):
                    df.at[index, 'Ultima_Actualizacion_Requerida'] = None
                else:
                    df.at[index, 'Ultima_Actualizacion_Requerida'] = f"CONVERT(DATETIME, '{row['Ultima_Actualizacion_Requerida']}', 120)"
            for index, row in df.iterrows():
                valores = [
                    "NULL" if pd.isna(valor) else f"'{valor}'"
                    for valor in row
                ]
                valores_sin_comillas = [v.strip("'") if v.startswith("'CONVERT") and v.endswith(")'") else v for v in valores]       
                columnas = ', '.join([f'{col}' for col in df.columns])
                try:
                    query = f"INSERT INTO dbo.Fact_Metadatos_Tablas ({columnas}) VALUES ({', '.join(valores_sin_comillas)})"
                    cursor_final.execute(query)
                    conn_final.commit()
                except Exception as e:
                    print(f"Error en la estructura de archivo de metadatos {esquema}.{nombre_tabla}: {e}")
                    log.write(f"{time.strftime('%Y-%m-%d %H:%M:%S')} - ERROR: En la estructura de archivo de metadatos {esquema}.{nombre_tabla}: {e}\n")

    except Exception as e:
        print(f"Error procesando la base de datos {db_name}: {e}")
        log.write(f"{time.strftime('%Y-%m-%d %H:%M:%S')} - ERROR: Procesando la base de datos {db_name}: {e}\n")

    Fin del proceso¶

    Finalmente, se registra el fin del proceso y se guarda el log.

    #tiempo total
end_time = time.time()
total_duration = end_time - start_time
print("Tiempo total de ejecución:", time.strftime("%H:%M:%S", time.gmtime(total_duration)))
log.write(f"{time.strftime('%Y-%m-%d %H:%M:%S')} - INFO: Tiempo total de ejecución: {time.strftime('%H:%M:%S', time.gmtime(total_duration))}\n")


#Cerrar log
log.close()
