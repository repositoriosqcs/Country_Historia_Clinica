    02 Poblar Bases Paramétricas Actualización¶

    El objetivo de este script es actualizar el repositorio centralizado a partir de las bases COL_CTY y CTY_PRI. La frecuencia sugerida es mensual y está a cargo del equipo de analítica de Country.

    Librerías¶

    Para empezar se importan las librerías requeridas, las cadenas de conexión connections_start y connections_final, y las funciones db_final y lista_tablas de Funciones Generales 

    import pyodbc
import pandas as pd
import os
import time
from funciones_generales import connections_start,connections_final,db_final,lista_tablas

    En esta parte se identifica si existe la carpeta para guardar los logs. Si no existe la crea automáticamente. Además, se da inicio al proceso.

output_dir = os.path.join(os.getcwd(), "logs","02.Poblar_Bases_Actualizacion")

# Limpiar los archivos de log al inicio del proceso

# Definir rutas y abrir los archivos de log (modo append)

# Tiempo de inicio del proceso

    Poblar tablas¶

    Para actualizar las tablas, se consultan las tablas definidas para el repositorio en las bases de origen COL_CTY y CTY_PRI y se llevan a las bases de destino COL_CTY_PARAMETRICA y CTY_PRI_PARAMETRICA.

    # Conectar a las bases y poblar tablas
esquema = 'dbo'
for db_name, conn_str in connections_start.items():
    print(f"Procesando base de datos: {db_name}")
    log.write(f"{time.strftime('%Y-%m-%d %H:%M:%S')} - INFO: Procesando base de datos: {db_name}\n")
    #print(conn_str)
    try:
        # Conexion a base fuente
        conn = pyodbc.connect(conn_str)
        cursor = conn.cursor()

        #Conexion a base destino

        db_name_final = db_final(db_name)
        conn_str_final = connections_final[db_name_final]
        conn_final = pyodbc.connect(conn_str_final)
        cursor_final = conn_final.cursor() 
        log.write(f"{time.strftime('%Y-%m-%d %H:%M:%S')} - INFO: Conexión a {db_name_final} exitosa \n")

        tablas_filtradas = lista_tablas(db_name_final,cursor_final)
        for nombre_tabla in tablas_filtradas:
            query = f"SELECT * FROM dbo.{nombre_tabla};"
            cursor.execute(query)
            #print(nombre_tabla)

            # Obtener los resultados de la consulta
            rows = cursor.fetchall()
            columns = [column[0] for column in cursor.description]  # Obtener los nombres de las columnas


            try:
                clean_query = f"DELETE FROM dbo.{nombre_tabla};"
                cursor_final.execute(clean_query)
                conn_final.commit()
                print(f"Tabla dbo.{nombre_tabla} limpiada exitosamente")
                log.write(f"{time.strftime('%Y-%m-%d %H:%M:%S')} - INFO: Tabla dbo.{nombre_tabla} limpiada exitosamente\n")
                #sql_script += f"-- Limpiar tabla {nombre_tabla_sql}\n" + clean_query + "\n"
            except Exception as e:
                print(f"Error al limpiar la tabla dbo.{nombre_tabla}: {e}")
                log.write(f"{time.strftime('%Y-%m-%d %H:%M:%S')} - ERROR: Al limpiar la tabla dbo.{nombre_tabla}: {e}\n")

            try:
                # Construir una consulta de inserción dinámica para cada fila
                for row in rows:
                    values_placeholder = ", ".join(["?"] * len(columns))  # Generar placeholders para los valores
                    insert_query = f"INSERT INTO {esquema}.{nombre_tabla} ({', '.join(columns)}) VALUES ({values_placeholder})"
                    cursor_final.execute(insert_query, row)
                    conn_final.commit()

                # Confirmar los cambios en la base de datos destino
                conn_final.commit()
                print(f"Datos de la tabla {nombre_tabla} transferidos a {db_name_final}.")
                log.write(f"{time.strftime('%Y-%m-%d %H:%M:%S')} - INFO: Datos de la tabla {nombre_tabla} transferidos a {db_name_final}.\n")
            except Exception as e:
                print(f"Error enviando la tabla {nombre_tabla} a {db_name_final}: {e}")
                log.write(f"{time.strftime('%Y-%m-%d %H:%M:%S')} - ERROR: Enviando la tabla {nombre_tabla} a {db_name_final}: {e}\n")

    except Exception as e:
        print(f"Error procesando la base de datos {db_name}: {e}")
        log.write(f"{time.strftime('%Y-%m-%d %H:%M:%S')} - ERROR: Procesando la base de datos {db_name}: {e}\n")

    Funciones para metadatos¶

    Para la extracción de los metadatos se definen las siguientes funciones que permiten hacer las consultas directamente en el motor SQL:

    # Consulta para los metadatos de las tablas
def consulta_metadatos_tablas(db_name):
    query_metadatos = f"""WITH Tablas as (    
SELECT
    TABLE_CATALOG AS Base,
    TABLE_SCHEMA AS Esquema,
    TABLE_NAME AS Tabla
FROM 
    {db_name}.INFORMATION_SCHEMA.TABLES
WHERE TABLE_SCHEMA = 'dbo' ),
TablasConDescripcion AS (
    SELECT
        obj.name AS Tabla,
        ep.name AS Propiedad,
        --ep.value AS Descripcion_Tabla,
        CAST(ep.value AS NVARCHAR(MAX)) AS Descripcion_Tabla,
        '{db_name}' as Base
    FROM
        {db_name}.sys.extended_properties AS ep
        INNER JOIN {db_name}.sys.objects AS obj ON ep.major_id = obj.object_id
    WHERE
        obj.schema_id = SCHEMA_ID('dbo')  -- Cambia 'dbo' por el esquema que necesites
        AND ep.minor_id = 0  -- Indica la tabla
        AND ep.name = 'MS_Description'
),
ActualizacionRegistros AS (
SELECT 
    DB_NAME(s.database_id) AS Base,
    t.name AS Tabla,
    s.last_user_update 
FROM 
    {db_name}.sys.tables t
LEFT JOIN 
    {db_name}.sys.dm_db_index_usage_stats s
ON 
    t.object_id = s.object_id
),
DataOwner AS (
SELECT
        obj.name AS Tabla,
        ep.name AS Propiedad,
        --ep.value AS DataOwner,
        CAST(ep.value AS NVARCHAR(MAX)) AS DataOwner,
        '{db_name}' as Base
    FROM
        {db_name}.sys.extended_properties AS ep
        INNER JOIN {db_name}.sys.objects AS obj ON ep.major_id = obj.object_id
    WHERE
        obj.schema_id = SCHEMA_ID('dbo')  -- Cambia 'dbo' por el esquema que necesites
        AND ep.minor_id = 0  -- Indica la tabla
        AND ep.name = 'DataOwner'
),
ContactEmail AS (
    SELECT
        obj.name AS Tabla,
        ep.name AS Propiedad,
        --ep.value AS ContactEmail,
        CAST(ep.value AS NVARCHAR(MAX)) AS ContactEmail,
        '{db_name}' as Base
    FROM
        {db_name}.sys.extended_properties AS ep
        INNER JOIN {db_name}.sys.objects AS obj ON ep.major_id = obj.object_id
    WHERE
        obj.schema_id = SCHEMA_ID('dbo')  -- Cambia 'dbo' por el esquema que necesites
        AND ep.minor_id = 0  -- Indica la tabla
        AND ep.name = 'ContactEmail'
),
FechasMetadatos AS (
    SELECT 
        '{db_name}' as Base,
        name AS Tabla,
        create_date,
        modify_date
    FROM 
        {db_name}.sys.objects
    WHERE 
        type = 'U'  -- Esto filtra solo las tablas de usuario
        AND schema_id = SCHEMA_ID('dbo')
)
SELECT 
    t.Base,
    t.Esquema,
    t.Tabla,
    td.Descripcion_Tabla,
    fm.create_date,
    fm.modify_date,
    ar.last_user_update,
    do.DataOwner,
    ce.ContactEmail
FROM Tablas t
LEFT JOIN TablasConDescripcion td ON t.Base = td.Base and t.Tabla = td.Tabla
LEFT JOIN ActualizacionRegistros ar ON t.Base = ar.Base and t.Tabla = ar.Tabla
LEFT JOIN DataOwner do ON t.Base = do.Base and t.Tabla = do.Tabla
LEFT JOIN ContactEmail ce ON t.Base = ce.Base and t.Tabla = ce.Tabla
LEFT JOIN FechasMetadatos fm ON t.Base = fm.Base and t.Tabla = fm.Tabla
INNER JOIN {db_name}_PARAMETRICA.sys.tables t_p ON t.Tabla = t_p.name
ORDER BY t.Tabla
"""
    return query_metadatos

    #Consulta para los metadatos de los campos
def consulta_metadatos_campos(db_name):
    query_metadatos = f"""WITH Columnas AS (
SELECT 
    TABLE_CATALOG AS Base,
    TABLE_SCHEMA AS Esquema,
    TABLE_NAME AS Tabla,
    COLUMN_NAME AS Columna,
    DATA_TYPE,
    CHARACTER_MAXIMUM_LENGTH
FROM 
    {db_name}.INFORMATION_SCHEMA.COLUMNS
WHERE TABLE_SCHEMA = 'dbo'
),
ColumnasConDescripcion AS (
    SELECT
        ep.name AS Propiedad,
        --ep.value AS Descripcion_Campo,
        CAST(ep.value AS NVARCHAR(MAX))AS Descripcion_Campo,
        obj.name AS Tabla,
        col.name AS Columna,
        '{db_name}' as Base
        --typ.name AS TipoDato,
    FROM
        {db_name}.sys.extended_properties AS ep
        INNER JOIN {db_name}.sys.objects AS obj ON ep.major_id = obj.object_id
        INNER JOIN {db_name}.sys.columns AS col ON ep.major_id = col.object_id AND ep.minor_id = col.column_id
        INNER JOIN {db_name}.sys.types AS typ ON col.user_type_id = typ.user_type_id
),
FechasMetadatos AS (
    SELECT 
        '{db_name}' as Base,
        obj.name AS Tabla,
        col.name AS Columna,
        obj.create_date,
        obj.modify_date
    FROM 
        {db_name}.sys.objects obj
    INNER JOIN 
        {db_name}.sys.columns col ON obj.object_id = col.object_id
    WHERE 
        obj.schema_id = SCHEMA_ID('dbo')  -- Ajusta el esquema si es necesario
)
SELECT 
    c.Base,
    c.Esquema,
    c.Tabla,
    c.Columna as Campo,
    cd.Descripcion_Campo,
    c.DATA_TYPE as DataType,
    c.CHARACTER_MAXIMUM_LENGTH as Longitud,
    fm.create_date,
    fm.modify_date
FROM Columnas c
LEFT JOIN ColumnasConDescripcion cd ON c.Base = cd.Base and c.Tabla = cd.Tabla and c.Columna = cd.Columna
LEFT JOIN FechasMetadatos fm ON c.Base = fm.Base and c.Tabla = fm.Tabla and c.Columna = fm.Columna
INNER JOIN {db_name}_PARAMETRICA.sys.tables t_p ON c.Tabla = t_p.name
order by c.tabla
"""
    return query_metadatos


    # Consulta para los metadatos de las llaves foráneas 
def consulta_metadatos_relaciones(db_name):
    query_metadatos = f"""SELECT         
               pk_tab.name AS Tabla_Primaria,
               pk_col.name AS Columna_Primaria,
               fk_tab.name AS Tabla_Foranea, 
               fk_col.name AS Columna_Foranea, 
               fk.name AS NombreRestriccion
        FROM {db_name}.sys.foreign_keys fk
        INNER JOIN {db_name}.sys.foreign_key_columns fkc ON fk.object_id = fkc.constraint_object_id
        INNER JOIN {db_name}.sys.tables fk_tab ON fk_tab.object_id = fkc.parent_object_id
        INNER JOIN {db_name}.sys.columns fk_col ON fk_col.column_id = fkc.parent_column_id AND fk_col.object_id = fk_tab.object_id
        INNER JOIN {db_name}.sys.tables pk_tab ON pk_tab.object_id = fkc.referenced_object_id
        INNER JOIN {db_name}.sys.columns pk_col ON pk_col.column_id = fkc.referenced_column_id AND pk_col.object_id = pk_tab.object_id
        INNER JOIN {db_name}_PARAMETRICA.sys.tables t_p ON pk_tab.name = t_p.name
"""
    return query_metadatos

    Procesamiento de metadatos¶

    Para la información de los metadatos se hacen las consultas a partir de las funciones anteriores en las bases de origen COL_CTY y CTY_PRI y los resultados se llevan a las bases de destino COL_CTY_PARAMETRICA y CTY_PRI_PARAMETRICA.

    for tabla in tablas_fuente.keys():
    #Conexion a base destino
    db_name = tablas_base[tabla]  
    db_name_final = db_final(db_name)
    conn_str_final = connections_final[db_name_final]
    conn_final = pyodbc.connect(conn_str_final)
    cursor_final = conn_final.cursor()
    log.write(f"{time.strftime('%Y-%m-%d %H:%M:%S')} - INFO: Conexión a base {db_name_final} realizada con éxito\n")

    print(tabla)
    try:
        clean_query = f"DELETE FROM {tablas_final[tabla]};"
        cursor_final.execute(clean_query)
        conn_final.commit()
        print(f"Tabla {tablas_final[tabla]} limpiada exitosamente")

    except Exception as e:
        print(f"Error al limpiar la tabla {tablas_final[tabla]}: {e}")
        log.write(f"{time.strftime('%Y-%m-%d %H:%M:%S')} - ERROR: Al limpiar la tabla {tablas_final[tabla]}: {e}\n")

    # Obtener los resultados de la consulta
    query = tablas_fuente[tabla]
    cursor_final.execute(query)
    rows = cursor_final.fetchall()
    columns = [column[0] for column in cursor_final.description]  # Obtener los nombres de las columnas    


    try:
        # Construir una consulta de inserción dinámica para cada fila      
        for row in rows:
            values_placeholder = ", ".join(["?"] * len(columns))  # Generar placeholders para los valores
            insert_query = f"INSERT INTO {tablas_final[tabla]} ({', '.join(columns)}) VALUES ({values_placeholder})"
            cursor_final.execute(insert_query, row)
            conn_final.commit()

            # Confirmar los cambios en la base de datos destino
            conn_final.commit()
        print(f"Datos de la tabla {tablas_final[tabla]} transferidos a {db_name_final}.")
        log.write(f"{time.strftime('%Y-%m-%d %H:%M:%S')} - INFO: Datos de la tabla {tablas_final[tabla]} transferidos a {db_name_final}.\n")
    except Exception as e:
        print(f"Error enviando la tabla {tablas_final[tabla]} a {db_name_final}: {e}")
        log.write(f"{time.strftime('%Y-%m-%d %H:%M:%S')} - ERROR: Enviando la tabla {tablas_final[tabla]} a {db_name_final}: {e}\n")

    Fin del proceso¶

    Finalmente, se registra el fin del proceso y se guarda el log.

    #tiempo total
end_time = time.time()
total_duration = end_time - start_time
print("Tiempo total de ejecución:", time.strftime("%H:%M:%S", time.gmtime(total_duration)))
log.write(f"{time.strftime('%Y-%m-%d %H:%M:%S')} - INFO: Tiempo total de ejecución: {time.strftime('%H:%M:%S', time.gmtime(total_duration))}\n")

#Cerrar log
log.close()
