    01 Creación Bases Repositorio¶

    El objetivo de este script es generar el SQL que permite crear la base de datos paramétrica. Es importante tener en cuenta que este script se debe ejecutar solo una vez para la instalación y no es necesario para la actualización.

    Librerías¶

    Para empezar se importan las librerías requeridas, las cadenas de conexión connections_start y connections_final, y las funciones db_final y lista_tablas de Funciones Generales

    import pyodbc
import pandas as pd
import os
import time
from funciones_generales import connections_start,connections_final,db_final,lista_tablas

    Scripts para tablas de metadatos y tablas adicionales para el tablero¶

    En cuanto a los metadatos se crean las tablas metadata_tablas, metadata_campos y metadata_relaciones que contienen los metadatos de las tablas, campos y llaves foráneas, respectivamente.     Funciones particulares¶

    Para este proceso en particular se definen las siguientes funciones:

    ## Guardar el script SQL generado
def guardar_script(nombre_archivo, contenido):
    ruta = os.path.join(destino, nombre_archivo)
    with open(ruta, "w", encoding="latin1") as f:       ## Codificación latin1
        f.write(contenido)
    print(f"Script {nombre_archivo} generado en {ruta}.")
    #log.write(f"{time.strftime('%Y-%m-%d %H:%M:%S')} - Script {nombre_archivo} generado en {ruta}.\n")

    ## Permite extraer la estructura de la tabla. Esto son su campos y tipo de dato para crear el script SQL
def estructura_tabla(cursor, esquema, tabla):
    # Consulta para obtener la estructura de la tabla
    try:
        query = f"""
            SELECT COLUMN_NAME, DATA_TYPE, CHARACTER_MAXIMUM_LENGTH, IS_NULLABLE
            FROM INFORMATION_SCHEMA.COLUMNS
            WHERE TABLE_NAME = '{tabla}'
            """
        cursor.execute(query)

        # Generar el script CREATE TABLE
        create_table_script = f"CREATE TABLE {esquema}.{tabla} (\n"
        for row in cursor.fetchall():
            column_name = row.COLUMN_NAME
            data_type = row.DATA_TYPE
            max_length = row.CHARACTER_MAXIMUM_LENGTH
            is_nullable = row.IS_NULLABLE

            # Construir la definición de columna
            column_definition = f"  {column_name} {data_type}"

            # Agregar tamaño si el tipo de dato lo requiere
            if max_length == -1:  # Longitud ilimitada
                column_definition += f"(MAX)"
            elif max_length and data_type in ('varchar', 'char', 'nvarchar', 'binary', 'varbinary'):
                column_definition += f"({max_length})"
            elif data_type in ('decimal', 'numeric'):  # Decimales requieren precisión y escala
                column_definition += "(18, 2)"  # Longitud predeterminada, ajustable según necesidad

            # Manejar NULL/NOT NULL
            if is_nullable == "NO":
                column_definition += " NOT NULL"
            else:
                column_definition += " NULL"

            # Añadir columna al script
            create_table_script += column_definition + ",\n"

        # Eliminar la coma final y cerrar la definición
        create_table_script = create_table_script.rstrip(",\n") + "\n)"
        return create_table_script
    except Exception as e:
        print(f"Error generando la estructura para la tabla {tabla}: {e}")
        log.write(f"{time.strftime('%Y-%m-%d %H:%M:%S')} - Error generando la estructura para la tabla {tabla}: {e}\n")
        return None

    ## Para crear la tabla se valida si existe una tabla y la elimina. Esto permite tener la estructura más reciente
def eliminar_tabla(esquema, tabla):
    # Generar el script SQL para eliminar la tabla si existe
    try:
        query = f"""IF OBJECT_ID('{esquema}.{tabla}', 'U') IS NOT NULL
        BEGIN
            DROP TABLE {esquema}.{tabla};
        END
        GO
        """
        log.write(f"{time.strftime('%Y-%m-%d %H:%M:%S')} - Script para eliminar {tabla} incorporado\n")
        return query
    except Exception as e:
        print(f"Error generando el script para eliminar la tabla {tabla}: {e}")
        log.write(f"{time.strftime('%Y-%m-%d %H:%M:%S')} - Error generando el script para eliminar la tabla {tabla}: {e}\n")
        return None

    Proceso principal¶

    Se define la conexión a las bases de origen con las cadenas de conexión definidas en connections_start 

    # Define conexión para las bases de datos, utilizando cadenas raw para evitar errores de escape
connections = connections_start

    Lectura de tablas seleccionadas para el repositorio¶

    Se lee el archivo de Tablas Únicas con el listado de tablas a incluir en el repositorio 

    # Definir ruta de carpeta con tablas para repositorio
ruta_excel = os.path.join(PARENT_DIR , "01.Fuentes","01.Tablas_Unicas.xlsx")

# Leer el archivo Excel
df = pd.read_excel(ruta_excel)

# Filtrar las tablas donde En_Repositorio es "SI"
tablas_filtradas = df[df['En_Repositorio'] == 'SI']['Tabla'].tolist()

    Iteración por tabla y consolidación del script SQL¶

    Para cada base de datos en la cadena de conexión de las bases de origen se genera el script SQL de las tablas seleccionadas. Además se incluyen las tablas de metadatos y las adicionales que necesita el tablero. 

    # Conectar a las bases y generar scripts
esquema = 'dbo'
for db_name, conn_str in connections.items():
    print(f"Procesando base de datos: {db_name}")
    log.write(f"{time.strftime('%Y-%m-%d %H:%M:%S')} - Procesando base de datos: {db_name}\n")
    try:
        conn = pyodbc.connect(conn_str)
        cursor = conn.cursor()
        log.write(f"{time.strftime('%Y-%m-%d %H:%M:%S')} - Conexión exitosa a {db_name}\n")

        # Generar el script para la base de datos
        script_db = ""
        db_name_parametrica = db_name + "_PARAMETRICA"
        script_db += f"USE {db_name_parametrica}\nGO\nSET ANSI_NULLS ON\nGO\nSET QUOTED_IDENTIFIER ON\nGO\n"

        for tabla in tablas_filtradas:
            # Eliminar la tabla si existe
            script_eliminar = eliminar_tabla(esquema, tabla)
            if script_eliminar:
                script_db += script_eliminar + "\nGO\n"

            # Crear la tabla
            script_tabla = estructura_tabla(cursor, esquema, tabla)
            if script_tabla:
                script_db += script_tabla + " ON [PRIMARY]\nGO\n"
                log.write(f"{time.strftime('%Y-%m-%d %H:%M:%S')} - Script para crear {tabla} incorporado\n")

        ##Incluir setencias de metadatos   
        script_db += eliminar_tabla(esquema, "Metadata_Tablas")
        script_db += metadata_tablas + "\nGO\n"

        script_db += eliminar_tabla(esquema, "Metadata_Campos")
        script_db += metadata_campos + "\nGO\n"

        script_db += eliminar_tabla(esquema, "Metadata_Relaciones")        
        script_db += metadata_relaciones + "\nGO\n"

        log.write(f"{time.strftime('%Y-%m-%d %H:%M:%S')} - Script para crear metadatos incorporado\n")

        ##Incluir setencias de tablas adicionales para el tablero

        script_db += eliminar_tabla(esquema, "Dim_Componente_Calidad")        
        script_db += dim_componente_calidad + "\nGO\n"

        script_db += eliminar_tabla(esquema, "Fact_Metadatos_Tablas")        
        script_db += fact_metadatos_tablas + "\nGO\n"

        script_db += eliminar_tabla(esquema, "Fact_Metadatos_Campos")        
        script_db += fact_metadatos_campos + "\nGO\n"

        script_db += eliminar_tabla(esquema, "Fact_Indicadores_Historicos")        
        script_db += fact_indicadores_historicos + "\nGO\n"

        log.write(f"{time.strftime('%Y-%m-%d %H:%M:%S')} - Script para crear tablas para tablero incorporado\n")             


        # Guardar el script en un archivo
        archivo_script = f"01.{db_name}_PARAMETRICA.sql"
        ruta_archivo = os.path.join(destino, archivo_script)
        with open(ruta_archivo, "w", encoding="latin1") as file:
            file.write(script_db)

        print(f"Script generado para {db_name}: 01.{db_name}_PARAMETRICA.sql")
        log.write(f"{time.strftime('%Y-%m-%d %H:%M:%S')} - Script generado para {db_name}: 01.{db_name}_PARAMETRICA.sql\n")
        conn.close()

    except Exception as e:
        print(f"Error procesando la base de datos {db_name}: {e}")
        log.write(f"{time.strftime('%Y-%m-%d %H:%M:%S')} - Error procesando la base de datos {db_name}: {e}\n")

print("¡Scripts de creación generados con éxito!")
log.write(f"{time.strftime('%Y-%m-%d %H:%M:%S')} -¡Scripts de creación generados con éxito!\n")

    Fin del proceso¶

    Finalmente, se registra el fin del proceso y se guarda el log. 

    #tiempo total
end_time = time.time()
total_duration = end_time - start_time
print("Tiempo total de ejecución:", time.strftime("%H:%M:%S", time.gmtime(total_duration)))
log.write(f"{time.strftime('%Y-%m-%d %H:%M:%S')} - Tiempo total de ejecución: {time.strftime('%H:%M:%S', time.gmtime(total_duration))}\n")

#Cerrar log
log.close()

    Tiempos de procesamiento¶

    Con un equipo con 32 GB de RAM y procesador 11th Gen Intel(R) Core(TM) i7-1165G7 @ 2.80GHz 1.69 GHz, el tiempo de ejecución promedio es de 20 segundos.
